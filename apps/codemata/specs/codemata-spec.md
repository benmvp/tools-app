## Overview

### Vision

Codemata provides best-in-class online code transformation tools that developers can bookmark and use daily. Each tool is fast, accurate, and enhanced with AI-generated educational content for SEO and user engagement.

### Key Features

- **Server-Side Transformations:** All code processing happens via Next.js Server Actions for security and consistency
- **AI-Enhanced Content:** Comprehensive educational content generated by Gemini 2.0 Flash
- **SEO Optimized:** Dynamic OpenGraph images, structured data, and rich metadata
- **Fast Navigation:** Command menu (⌘K) with fuzzy search across all tools
- **Dark Mode:** System-aware with localStorage persistence
- **Mobile Responsive:** Optimized for both desktop and mobile workflows
- **Accessible:** WCAG 2.0/2.1 Level AA compliant

## Architecture Patterns

### Development Ports

- **Dev server:** `localhost:3001` (`pnpm dev`)
- **Production build:** `localhost:3333` (for E2E/Lighthouse testing with `pnpm build && pnpm start`)

### Server Action Pattern

All transformations follow this pattern:

```typescript
// app/formatters/actions.ts
'use server'

import { format } from 'prettier'
import type { FormatConfig } from '@/lib/types'

export async function formatTypescriptAction(
  input: string,
  config: FormatConfig,
): Promise<string> {
  const indent = getIndentSize(config.indentation)

  try {
    const formatted = await format(input, {
      parser: 'typescript',
      tabWidth: indent.size,
      useTabs: indent.useTabs,
      semi: true,
      singleQuote: true,
      trailingComma: 'es5',
    })

    return formatted
  } catch (error) {
    throw new Error(`Formatting failed: ${error.message}`)
  }
}
```

### Dynamic Page Pattern

```typescript
// app/formatters/[slug]/page.tsx
import { notFound } from 'next/navigation'
import { Transformer } from '@/components/Transformer'
import { FormatterAIContent } from '@/components/FormatterAIContent'
import { FORMATTER_TOOLS } from '@/lib/tools-data'
import { getFormatterContent } from '@/lib/ai/helpers'

export const revalidate = 86400 // 24-hour ISR

export function generateStaticParams() {
  return Object.keys(FORMATTER_TOOLS).map((slug) => ({ slug }))
}

export async function generateMetadata({ params }: Props) {
  const tool = FORMATTER_TOOLS[params.slug]
  if (!tool) return notFound()

  const aiContent = await getFormatterContent(params.slug, tool.name)

  return {
    title: aiContent?.seo.title || tool.metadata.title,
    description: aiContent?.seo.description || tool.metadata.description,
    // ... OpenGraph, Twitter, etc.
  }
}

export default async function Page({ params }: Props) {
  const tool = FORMATTER_TOOLS[params.slug]
  if (!tool) return notFound()

  const aiContent = await getFormatterContent(params.slug, tool.name)

  return (
    <>
      <CategoryBackLink href="/formatters" label="Formatters" />
      <h1>{tool.name}</h1>

      {aiContent && <FormatterIntro intro={aiContent.intro} />}

      <Transformer
        action={tool.action}
        example={tool.example}
        language={tool.language}
      />

      {aiContent && <FormatterAIContent content={aiContent} />}
    </>
  )
}
```

### AI Content Generation

Content generated server-side with ISR:

```typescript
// lib/ai/helpers.ts
export async function getFormatterContent(
  slug: string,
  toolName: string,
): Promise<FormatterContent | undefined> {
  if (!shouldGenerateAI()) return undefined

  const cacheKey = `formatter-${slug}`
  const cached = getFromCache(cacheKey)
  if (cached) return cached

  try {
    const systemPrompt = getFormatterSystemPrompt()
    const userPrompt = getFormatterUserPrompt(toolName, getAllToolNames())

    const content = await generateWithGemini(systemPrompt, userPrompt)
    const validated = formatterContentSchema.parse(content)

    setCache(cacheKey, validated)
    return validated
  } catch (error) {
    console.error('AI generation failed:', error)
    return undefined // Graceful degradation
  }
}
```
